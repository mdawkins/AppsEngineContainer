podman build --build-arg RH_VERSION=10 -t alm:npp -f ./Dockerfile.npp .

podman run -d \
	--name aeweb \
	-p 443:443 \
	-v ~/Containers/wwwroot:/var/www/html:z \
	-v ~/Containers/AppsEngineContainer/localhost.crt:/etc/pki/tls/certs/localhost.crt:z \
	-v ~/Containers/AppsEngineContainer/localhost.key:/etc/pki/tls/private/localhost.key:z \
	-v ~/Containers/wwwroot/HAE_ConfigFiles/nginx_default.conf:/etc/nginx/conf.d/default.conf:z \
	alm:npp

podman system prune --all
podman stop aeweb && podman container rm aeweb
podman exec -it aeweb /bin/bash

openssl req -x509 -out localhost.crt -keyout localhost.key -newkey rsa:2048 -nodes -sha256 -subj '/CN=localhost'
