docker build --build-arg RH_VERSION=9 -t alm9:npp -f ./Dockerfile .

docker run -d \
	--name aeweb \
	-p 443:443 \
	-v /var/www/html:/var/www/html:ro \
	-v /etc/pki/tls/certs/localhost.crt:/etc/pki/tls/certs/localhost.crt:ro \
	-v /etc/pki/tls/private/localhost.key:/etc/pki/tls/private/localhost.key:ro \
	-v /var/www/html/HAE_ConfigFiles/nginx_default.conf:/etc/nginx/conf.d/default.conf \
	alm9:npp

docker stop aeweb && docker container rm aeweb

docker exec -it aeweb /bin/bash

docker exec -it aeweb ss -tlnp

openssl req -x509 -out localhost.crt -keyout localhost.key -newkey rsa:2048 -nodes -sha256 -subj '/CN=localhost'
